7. Использование языка

В настоящее время Lua используется в различных проектах, где требуется достаточно быстрый и нетрудный в освоении скриптовый язык программирования — например, в разработке игр, где Lua часто используется в качестве прослойки между игровым движком и данными для написания сценариев поведения и взаимодействия объектов. Благодаря компактности применим и в портативных устройствах, в частности, один из графических микрокалькуляторов Texas Instruments, а именно, TI-Nspire CX, использует язык LUA помимо традиционного для такого класса устройств Бейсика.

Пример скрипта на языке Lua (название кода: “Угадай число”):

-- Инициализация генератора случайных чисел
math.randomseed(os.time())

local target = math.random(1, 100) -- Загаданное число от 1 до 100
local attempts = 0
local guess = nil

print("--- Игра: Угадай число ---")
print("Я загадал число от 1 до 100. Попробуй отгадать!")

-- Цикл продолжается, пока игрок не угадает
while guess ~= target do
    io.write("Введи свой вариант: ")
    local input = io.read()      -- Читаем ввод пользователя
    guess = tonumber(input)      -- Преобразуем строку в число

    if not guess then
        print("Ошибка: Пожалуйста, введи число.")
    else
        attempts = attempts + 1  -- Инкремент попыток

        if guess < target then
            print("Больше!")
        elseif guess > target then
            print("Меньше!")
        else
            print("Поздравляю! Ты угадал за " .. attempts .. " попыток.")
        end
    end
end
