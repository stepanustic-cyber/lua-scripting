🔀 Ключевое слово: or

Логический оператор «ИЛИ» (or)

or — это логический оператор в языке Lua. Как и and, он возвращает один из своих операндов, а не булево значение. В Lua false и nil считаются ложными, всё остальное — истинным.

📌 Принцип работы:
• Если первый операнд true (не false и не nil) → возвращается первый операнд
• Если первый операнд false или nil → возвращается второй операнд
• Поддерживает «короткое замыкание»: второй операнд вычисляется только при необходимости

Примеры

-- Базовое использование с булевыми значениями print(true or true) --> true print(true or false) --> true (первый истинный, второй не вычисляется) print(false or true) --> true print(false or false) --> false
-- Возврат небулевых значений (особенность Lua) local x = nil local y = 20 local result = x or y print(result) --> 20 (nil считается ложным, возвращается y) local name = "Анна" local default = "Гость" print(name or default) --> Анна (name истинен, возвращается name) print(nil or default) --> Гость (nil ложен, возвращается default)
-- Использование в условных операторах local isWeekend = false local isHoliday = true if isWeekend or isHoliday then print("Можно отдохнуть!") else print("Рабочий день") end --> Можно отдохнуть!

Идиоматические паттерны

-- Значение по умолчанию (самый частый паттерн) local config = { timeout = nil } local timeout = config.timeout or 5000 print(timeout) --> 5000 (значение по умолчанию)
-- Тернарный оператор (вместе with and) -- x and y or z работает как тернарный оператор local age = 20 local status = (age >= 18) and "взрослый" or "ребёнок" print(status) --> взрослый local value = nil local result = value and "есть значение" or "нет значения" print(result) --> нет значения
-- Выбор первого не-nil значения из нескольких local a = nil local b = false local c = "значение" local d = "запасное" local first = a or b or c or d print(first) --> false (false не nil, поэтому он и вернулся) -- Внимание: false считается истинным для or local x = false local y = "default" print(x or y) --> false (false не nil и не false? false — false, так что вернётся y?) -- На самом деле: x = false → первый операнд ложный → возвращается y
-- Выбор первого не-nil и не-false значения local function firstTruthy(...) for i = 1, select("#", ...) do local val = select(i, ...) if val and val ~= false then return val end end return nil end print(firstTruthy(nil, false, "hello", 42)) --> hello

Важные особенности

⚠️ Приоритет операторов:
and имеет более высокий приоритет, чем or
-- Без скобок local a = true or false and false -- Эквивалентно: true or (false and false) → true -- Со скобками (другой порядок) local b = (true or false) and false -- (true or false) and false → true and false → false print(a, b) --> true false
⚠️ or возвращает значение, а не true/false
local x = 5 local y = 10 local result = x or y print(result) --> 5 (не true, а само число 5!) if x or y then print("Условие истинно") --> Это выполнится, т.к. 5 — истина end

Частые ошибки

❌ Путаница с false:
local userInput = false local defaultValue = "default" -- Ожидание: вернётся defaultValue (ведь false — это ложь) -- Реальность: вернётся false (потому что false — это законное значение) local result = userInput or defaultValue print(result) --> false, а не "default"! -- Правильное решение: явная проверка на nil local result2 = (userInput == nil) and defaultValue or userInput
❌ Забыто короткое замыкание:
-- Функция, которая может вызвать ошибку local function risky() error("Ошибка!") end -- Эта строка НЕ вызовет ошибку, потому что первый аргумент true local safe = true or risky() --> risky() не вызывается -- А эта вызовет, потому что первый аргумент false local unsafe = false or risky() --> Ошибка!
💡 Полезный паттерн: memoization (кэширование)
local function getData() local cached = nil return function() cached = cached or expensiveCalculation() return cached end end -- expensiveCalculation() вызовется только один раз
